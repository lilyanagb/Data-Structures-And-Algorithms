#include <iostream>
#include <unordered_map>
#include <vector>
#include <string>
using namespace std;


int main() {
    long count;
    cin >> count;
    string s;

    unordered_map<string, long> mp;
    vector<string> str;
    while(count--) {
        cin >> s;
       unordered_map<string, long>::iterator it = mp.find(s);

        if (it != mp.end()) {
            it->second++;
        }
        else {
            mp.insert(make_pair(s, 1));
        }
        str.push_back(s);
    }
    for (auto x : str) {
        cout << mp[x] << " ";
 }
    return 0;
}
