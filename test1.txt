#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


void countingSort(vector<int>& arr) {
    auto maxNumber = *max_element(arr.begin(), arr.end());
    int len = arr.size();
    vector<int> counter(maxNumber + 1);

    for (int i = 0; i < maxNumber + 1; i++) {
        counter[i] = 0;
    }

    for (int i = 0; i < len; i++) {
        counter[arr[i]]++;
    }

    int curr = 0;
    for (int i = 0; i <= maxNumber; i++) {

        for (int j = 0; j < counter[i]; j++) {
            arr[curr++] = i;
        }
    }
}
int main() {
    int count;
    cin >> count;
    int input;
    vector<int> arr;
    for (int i = 0; i < count; i++) {
        cin >> input;
        arr.push_back(input);
    }
    countingSort(arr);
    vector<int> noDouble;

    noDouble.push_back(arr[0]);
    for (int i = 0; i < count-1; i++) {
        if (arr[i] != arr[i + 1]) {
            noDouble.push_back(arr[i + 1]);
        }
    }

    for (size_t i = 0; i <noDouble.size(); i++) {
        cout << noDouble[i] << ' ';
    }
    return 0;
}

bool compare(const pair<pair<long, double>, long>& a, const pair<pair<long, double>, long>& b) {
    if (a.first.second == b.first.second) {
        return (a.first.first > b.first.first);
    }
    else {
        return (a.first.second > b.first.second);
    }
}
